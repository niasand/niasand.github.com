<pre class="lang:default decode:true">#include &lt;iostream&gt;
#include "constants.h"

float distance(float h,float x)
{
 using namespace std;
 float fallen;
 fallen = h - constants::gravity_constant * (x*x / 2);
 if (fallen&lt; 0)
 return false;
 else
 return fallen;
}

void printdistance()
{
 using namespace std;
