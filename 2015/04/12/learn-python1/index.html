import math
def jiefangcheng(a,b,c):
    delta = b*b - 4*a*c
    if delta &lt; 0:
       print "Error,The delta value is not right"       

    elif delta == 0:
        print "x =", (-b)/(2*a)
    else:
        deltap = math.sqrt(delta)
        x1 = (-b+ deltap)/(2*a)
        x2 = (b + deltap)/(2*a)
        print "x1= %.3f" % x1
        print "x2= %.3f" % x2

jiefangcheng(1,3,1)
